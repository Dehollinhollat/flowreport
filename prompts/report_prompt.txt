const rows = $input.all().map(item => item.json);

const total = rows.length;
const terminees = rows.filter(r => r.statut === "Terminé").length;
const bloquees = rows.filter(r => r.statut === "Bloqué").length;
const hautePrioNonTerminees = rows.filter(r => r.priorite === "Haute" && r.statut !== "Terminé").length;

const heuresParConsultant = {};
rows.forEach(r => {
  const h = parseFloat(r.heures) || 0;
  heuresParConsultant[r.consultant] = (heuresParConsultant[r.consultant] || 0) + h;
});

const totalHeures = Object.values(heuresParConsultant).reduce((a, b) => a + b, 0);
const tauxCompletion = ((terminees / total) * 100).toFixed(1);
const date = new Date().toLocaleDateString('fr-FR');

const tableauConsultants = Object.entries(heuresParConsultant)
  .map(([nom, h]) => `| ${nom} | ${h}h |`)
  .join('\n');

const prompt = `Tu es un assistant spécialisé dans le reporting de projets ESN. Génère un rapport d'activité hebdomadaire en HTML propre et professionnel basé sur ces KPIs :

## KPIs Globaux

| Métrique | Valeur |
|----------|--------|
| Taux de complétion | ${tauxCompletion}% |
| Total heures équipe | ${totalHeures}h |
| Tâches bloquées | ${bloquees} |
| Tâches haute priorité non terminées | ${hautePrioNonTerminees} |

## Heures par Consultant

| Consultant | Heures |
|------------|--------|
${tableauConsultants}

Le rapport HTML doit :
Le rapport HTML doit être compatible Gmail avec ces contraintes techniques strictes :
- Utiliser des tableaux HTML pour la mise en page (pas de flexbox ni CSS grid)
- Utiliser uniquement des styles inline (pas de classes CSS externes)
- Utiliser des couleurs de fond solides (pas de linear-gradient)
- Cartes KPIs : fond vert (#27AE60) pour taux complétion et heures, fond orange (#E67E22) pour tâches bloquées, fond rouge (#C0392B) pour haute priorité non terminées, texte blanc en gras
- Barres de progression : simulées avec des tableaux imbriqués et des cellules colorées
- Header : fond #1F4E79 texte blanc centré avec titre et date
- Footer : fond #1F4E79 texte blanc centré avec le nom du préparateur
- Toutes les sections avec bordure gauche colorée et fond légèrement teinté
- Police Arial ou sans-serif uniquement
- Inclure un résumé global, les points d'attention et les recommandations. Rien d’autre.
- Être responsive et lisible sur mobile
- Retourne uniquement le code HTML complet, sans markdown, sans balises de code
- Le rapport doit être COMPLET et se terminer obligatoirement par le footer avec le nom du préparateur. Ne jamais couper le HTML à mi-chemin.
- Ne pas inclure de tableau récapitulatif des KPIs en fin de rapport, les KPIs sont déjà présentés en début de rapport


Date : ${date}
Préparé par : Déhollin HOLLAT, Chef de Projet Data IA Projet FlowReport`;

return [{
  json: {
    tauxCompletion: parseFloat(tauxCompletion),
    totalHeures,
    heuresParConsultant,
    tachesBloquees: bloquees,
    hautePrioNonTerminees,
    requestBody: {
      model: "claude-haiku-4-5-20251001",
      max_tokens: 8192,
      messages: [{ role: "user", content: prompt }]
    }
  }
}];
